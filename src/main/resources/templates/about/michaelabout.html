public static double sqrt(double X) {
	for (int i = 1; i < X; ++i) {
		int p = i * i;
		if (p == X) {
			// perfect square
			return i;
		}
		if (p > X) {
			// found left part of decimal
			return sqrt(X, i - 1, i);
		}
	    }
	return Double.NaN;

public static char longestStreakf(String a) {
    char answer = ' ';
    int streak = 1;
    int longestStreak = 1;

    for(int i = 0; i < a.length() - 1; i++) {

        if(a.charAt(i) == a.charAt(i + 1)) {
            streak++;
            if(streak > longestStreak) {
                longestStreak = streak;
                answer = a.charAt(i);
            }
        } else {
            streak = 1;
        }
        
    }

    System.out.println(longestStreak);
    return answer;
}
